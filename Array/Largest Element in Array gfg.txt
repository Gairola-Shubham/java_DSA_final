class Solution {
    public static int largest(int[] arr) {
        int largest = arr[0];
        for(int i=1; i<arr.length; i++){
            if(arr[i]>largest){
                largest = arr[i];
            }
        }
        return largest;
    }
    public static int largest1(int[] arr) {
        // code here
        int[] temp = arr.clone();
        Arrays.sort(temp);
        return temp [temp.length-1];
    }
}
